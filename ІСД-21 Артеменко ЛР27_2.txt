#include <iostream>
#include <stdexcept>
using namespace std;

// Функція ділення, кидає виняток при діленні на нуль
float internalDiv(float arg1, float arg2) {
    if (arg2 == 0.0) throw runtime_error("Your input is not valid. You can't divide by zero.");
    return arg1 / arg2;
}

int main() {
    float a, b;

    while (cin >> a >> b) {
        try {
            float r = internalDiv(a, b);
            cout << r << endl;
        } catch (runtime_error &e) {
            cout << e.what() << endl;
        }
    }

    return 0;
}
