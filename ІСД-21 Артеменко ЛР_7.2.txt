#include <iostream>
#include <iomanip>  // для встановлення точності
using namespace std;

int main(void) {
    double pi4 = 0.0; // тут зберігається наближення π/4
    long n;

    cout << "Number of iterations? ";
    cin >> n;

    // Обчислюємо за формулою Лейбніца
    for (long i = 0; i < n; i++) {
        // знак чергується: +, -, +, - ...
        double term = 1.0 / (2 * i + 1); 
        if (i % 2 == 0)
            pi4 += term;
        else
            pi4 -= term;
    }

    cout.precision(20); // встановлюємо точність виводу
    cout << "Pi = " << (pi4 * 4.0) << endl;

    return 0;
}
