#include <iostream>
using namespace std;

int main() {
    const int n = 3;
    int a[n][n];

    // Ввід матриці
    cout << "Enter elements of the matrix:\n";
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            cin >> a[i][j];

    int mainDiag = 0;     // Головна діагональ
    int secondDiag = 0;  // Побічна діагональ

    // Обчислення сум діагоналей
    for (int i = 0; i < n; i++) {
        mainDiag += a[i][i];
        secondDiag += a[i][n - 1 - i];
    }

    cout << "Main diagonal sum: " << mainDiag << endl;
    cout << "Secondary diagonal sum: " << secondDiag << endl;

    return 0;
}
