#include <iostream>   // Підключаємо бібліотеку для вводу/виводу
using namespace std;  // Щоб не писати std:: перед cout, cin, endl

int main() {
    // Змінна c0 для початкового числа (long long — щоб витримувати великі проміжні значення)
    long long c0; // Використовуємо long long, бо проміжні значення послідовності Коллатца можуть ставати дуже великими

    // Просимо користувача ввести число
    cout << "Enter the natural number: ";
    cin >> c0; // Зчитуємо значення у змінну c0

    // Перевірка коректності вводу: число повинно бути натуральним (позитивним)
    if (c0 <= 0) {
        cout << "Wrong number! You need to write natural number (> 0)." << endl;
        return 1; // Завершуємо програму з кодом 1 (помилка)
    }

    int steps = 0; // Лічильник кроків (скільки разів застосовано правило)

    // Основний цикл: виконуємо операції, доки c0 не стане рівним 1
    while (c0 != 1) {
        // Якщо c0 парне — ділимо на 2
        if (c0 % 2 == 0)
            c0 = c0 / 2;
        // Інакше (тобто якщо непарне) — множимо на 3 і додаємо 1
        else
            c0 = 3 * c0 + 1;

        // Виводимо поточне значення c0 (проміжний член послідовності)
        cout << c0 << " ";

        // Збільшуємо лічильник кроків на 1
        steps++;
    }

    // Після того, як цикл закінчився (тобто c0 == 1), виводимо кількість кроків
    cout << "steps = " << steps << endl;

    return 0; // Успішне завершення програми
}
