#include <iostream>
using namespace std;
int main(void) {
    int vector[] = { 3, -5, 7, 10, -4, 14, 5, 2, -13 };
    int n = sizeof(vector) / sizeof(vector[0]);
    int *ptr = vector;          // вказівник на початок масиву
    int *end = vector + n;      // вказівник на елемент після останнього
    int minValue = *ptr;        // початкове мінімальне значення
    for (; ptr < end; ptr++) {
        if (*ptr < minValue) {
            minValue = *ptr;
        }
    }
    cout << "Smallest element of array: " << minValue << endl;
    return 0;
}
