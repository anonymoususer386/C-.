#include <iostream>
using namespace std;

int main() {
    int n;   // Number of elements
    cout << "Enter the number of elements: ";
    cin >> n;

    int arr[n];   // Declare array

    // Input array elements
    cout << "Enter array elements:\n";
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    // Find index of the first maximum element
    int maxIndex = 0;
    for (int i = 1; i < n; i++) {
        if (arr[i] > arr[maxIndex]) {
            maxIndex = i;
        }
    }

    // Replace negative elements with 0 before the first maximum
    for (int i = 0; i < maxIndex; i++) {
        if (arr[i] < 0) {
            arr[i] = 0;
        }
    }

    // Output the result
    cout << "\nProcessed array:\n";
    for (int i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }

    return 0;   // End of program
}
