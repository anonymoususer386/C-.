#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    const int m = 12;
    int arr[m];

    // Input array
    cout << "Enter 12 integers:\n";
    for (int i = 0; i < m; i++) {
        cin >> arr[i];
    }

    // Sort ascending
    sort(arr, arr + m);
    cout << "\nSorted in ascending order:\n";
    for (int i = 0; i < m; i++) {
        cout << arr[i] << " ";
    }

    // Sort descending
    sort(arr, arr + m, greater<int>());
    cout << "\nSorted in descending order:\n";
    for (int i = 0; i < m; i++) {
        cout << arr[i] << " ";
    }

    return 0;
}
