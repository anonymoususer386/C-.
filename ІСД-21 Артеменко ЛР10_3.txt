#include <iostream>
using namespace std;

int main() {
    const int m = 5, n = 7;
    int a[m][n];

    // Ввід масиву
    cout << "Enter elements of the matrix:\n";
    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++)
            cin >> a[i][j];

    // Обчислення суми кожного рядка
    for (int i = 0; i < m; i++) {
        int sum = 0;
        for (int j = 0; j < n; j++) {
            sum += a[i][j];
        }
        cout << "Sum of row " << i << ": " << sum << endl;
    }

    return 0;
}
