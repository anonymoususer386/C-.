#include <iostream>
#include <string>
#include <cmath> // для abs
using namespace std;

// Допоміжна функція для знаходження НСД
int gcd(int a, int b) {
    a = abs(a);
    b = abs(b);
    while (b != 0) {
        int temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

class Fraction {
public:
    Fraction(int num, int denom) : numerator(num), denominator(denom) {
        if (denominator == 0) {
            cout << "Error: denominator cannot be zero!" << endl;
            denominator = 1;
        }
        reduce();
    }

    // Повертає рядок у вигляді дробу або мішаного дробу
    string toString() {
        int whole = numerator / denominator;
        int rem = abs(numerator % denominator);

        if (rem == 0) // ціле число
            return to_string(whole);
        else if (abs(numerator) > denominator) // мішаний дріб
            return to_string(whole) + " " + to_string(rem) + "/" + to_string(denominator);
        else // правильний дріб
            return to_string(numerator) + "/" + to_string(denominator);
    }

    // Повертає десяткове представлення
    double toDouble() {
        return double(numerator) / denominator;
    }

    // Порівняння
    bool isGreaterThan(Fraction that) {
        return this->numerator * that.denominator > that.numerator * this->denominator;
    }

    bool isLessThan(Fraction that) {
        return this->numerator * that.denominator < that.numerator * this->denominator;
    }

    bool isEqual(Fraction that) {
        return this->numerator * that.denominator == that.numerator * this->denominator;
    }

private:
    int numerator;
    int denominator;

    void reduce() {
        int g = gcd(numerator, denominator);
        numerator /= g;
        denominator /= g;

        // завжди робимо знаменник додатним
        if (denominator < 0) {
            denominator = -denominator;
            numerator = -numerator;
        }
    }
};

int main() {
    int n1, d1, n2, d2;
    char slash;

    cout << "Enter first fraction (e.g., 3 / 4): ";
    cin >> n1 >> slash >> d1;
    cout << "Enter second fraction (e.g., 1 / 3): ";
    cin >> n2 >> slash >> d2;

    Fraction f1(n1, d1);
    Fraction f2(n2, d2);

    string result = f1.toString();

    if (f1.isGreaterThan(f2))
        result += " > " + f2.toString();
    else if (f1.isLessThan(f2))
        result += " < " + f2.toString();
    else
        result += " = " + f2.toString();

    cout << result << endl;

    return 0;
}
