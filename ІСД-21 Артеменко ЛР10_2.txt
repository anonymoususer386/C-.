#include <iostream>
using namespace std;

int main() {
    const int m = 6, n = 4;
    int a[m][n];

    // Ввід масиву
    cout << "Enter elements of the matrix:\n";
    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++)
            cin >> a[i][j];

    // Підрахунок парних чисел у кожному стовпчику
    for (int j = 0; j < n; j++) {
        int countEven = 0;
        for (int i = 0; i < m; i++) {
            if (a[i][j] % 2 == 0)
                countEven++;
        }
        cout << "Even numbers in column " << j << ": " << countEven << endl;
    }

    return 0;
}
