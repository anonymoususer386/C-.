#include <iostream>
using namespace std;

int main() {
    const int m = 4, n = 6;
    int a[m][n];

    // Ввід масиву
    cout << "Enter elements of the matrix:\n";
    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++)
            cin >> a[i][j];

    // Обчислення суми кожного стовпчика
    for (int j = 0; j < n; j++) {
        int sum = 0;
        for (int i = 0; i < m; i++) {
            sum += a[i][j];
        }
        cout << "Sum of column " << j << ": " << sum << endl;
    }

    return 0;
}
