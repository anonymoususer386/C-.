#include <iostream>
#include <string>
using namespace std;

class Person
{
public:
    string name;
    int age;
    string city;     // нове поле
    double height;   // нове поле
};

void print(Person* person)
{
    cout << person->name << " is "
         << person->age << " years old, lives in "
         << person->city << ", height "
         << person->height << " cm"
         << endl;
}

int main()
{
    Person person1;
    person1.name = "Harry";
    person1.age = 23;
    person1.city = "London";
    person1.height = 180;

    Person person2;
    person2.name = "Emma";
    person2.age = 28;
    person2.city = "Paris";
    person2.height = 165;

    cout << "Meet " << person1.name << endl;
    print(&person1);

    cout << "Meet " << person2.name << endl;
    print(&person2);

    return 0;
}
