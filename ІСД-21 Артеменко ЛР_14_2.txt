#include <iostream>
#include <cstdlib>
#include <ctime>

using namespace std;

int main() {
    const int rows = 3;
    const int cols = 3;

    srand(time(0));

    // виділення пам’яті для рядків
    int** arr = new int*[rows];

    // виділення пам’яті для кожного рядка
    for (int i = 0; i < rows; i++) {
        arr[i] = new int[cols];
    }

    int even = 0, odd = 0;

    cout << "Matrix:\n";

    // заповнення та виведення
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            arr[i][j] = rand() % 50 + 1;
            cout << arr[i][j] << "\t";

            if (arr[i][j] % 2 == 0)
                even++;
            else
                odd++;
        }
        cout << endl;
    }

    cout << "\nMain diagonal: ";
    for (int i = 0; i < rows; i++) {
        cout << arr[i][i] << " ";
    }

    cout << "\nSecondary diagonal: ";
    for (int i = 0; i < rows; i++) {
        cout << arr[i][cols - 1 - i] << " ";
    }

    cout << "\n\nEven elements: " << even << endl;
    cout << "Odd elements: " << odd << endl;

    // звільнення пам’яті
    for (int i = 0; i < rows; i++) {
        delete[] arr[i];
    }

    delete[] arr;

    return 0;
}
