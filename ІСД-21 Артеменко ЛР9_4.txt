#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "Enter number of elements: ";
    cin >> n;

    int arr[n];

    // Input array
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    int A;
    cout << "Enter A: ";
    cin >> A;

    int count = 0;

    // Count elements greater than A
    for (int i = 0; i < n; i++) {
        if (arr[i] > A)
            count++;
    }

    cout << "Number of elements greater than A: " << count << endl;

    return 0;
}
