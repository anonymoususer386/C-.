#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "Enter number of elements: ";
    cin >> n;

    int arr[n];

    // Input array
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    int zeroIndex = -1;

    // Find first zero
    for (int i = 0; i < n; i++) {
        if (arr[i] == 0) {
            zeroIndex = i;
            break;
        }
    }

    if (zeroIndex == -1)
        cout << "Array does not contain zero elements." << endl;
    else
        cout << "Index of first zero element: " << zeroIndex << endl;

    return 0;
}
