#include <iostream>
#include <stdexcept>
using namespace std;

int main() {
    int a = 0, b = 0, c = 0;
    cin >> b >> a;

    try {
        if (b == 0) throw runtime_error("Your input is not valid, you can't divide by zero.");
        c = a / b;
    } catch (runtime_error &e) {
        cout << e.what() << endl;
        c = 0;
    }

    cout << c << endl;
    return 0;
}
